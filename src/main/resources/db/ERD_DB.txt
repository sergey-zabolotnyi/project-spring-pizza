Table category {
  id integer [primary key]
  category_en varchar(255)
  category_ru varchar(255)
}

Table user {
  id integer [primary key]
  email varchar(255)
  user varchar(255)
  password varchar(255)
  role varchar(255)
  time timestamp
}

Table orders {
  id integer [primary key]
  user_id integer
  totalPrice numeric(9,2)
  status varchar(255)
  time timestamp
}

Table dish {
  id integer [primary key]
  name_en varchar(255)
  name_ru varchar(255)
  price numeric(9,2)
  category_id integer
  time timestamp
}

Table cart {
  id integer [primary key]
  user_id integer
  dish_id integer
}

Table order_dish {
  id integer [primary key]
  order_id integer
  dish_id integer
}

Ref: orders.user_id > user.id // many-to-one
Ref: cart.user_id > user.id // many-to-one
Ref: cart.dish_id > dish.id // many-to-one
Ref: dish.category_id > category.id // many-to-one
Ref: order_dish.order_id > orders.id // many-to-one
Ref: order_dish.dish_id > dish.id // many-to-one
